Examples




MyComponents.yml

Create components:

apiVersion: core.oam.dev/v1alpha2
kind: Component
metadata:
  name: web
spec:
  type: Server
  settings:
    containers:
      - image: oamdev/helloworld-python:v1
        ports:
          - protocol: TCP
            containerPort: 9999
            name: http
---
apiVersion: core.oam.dev/v1alpha2
kind: Component
metadata:
  name: function
spec:
  type: Function
  settings:
    runtime: java8
    blobCodeUrl: http://xxx/f36286e6-f070-498f-a520-6dd3a589bd90

MyAppConfig.yml

Assign operations to components:

apiVersion: core.oam.dev/v1alpha2
kind: ApplicationConfiguration
metadata:
  name: myapp
spec:
  components:
    - componentName: web
      traits:
        - name: autoscaler
          properties:
            min: 1
            max: 100
        - name: rollout
          properties:
            canary:
              maxSurge: 25%
              maxUnavailable: 0
              steps:
                - setWeight: 10
    - componentName: function
      traits:
        - name: autoscaler
          properties:
            min: 0
            max: 1000
            triggers:
              - type: metaq
